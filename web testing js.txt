// Email validation
function validateEmail() {
    let email = document.getElementById("email").value;
    let pattern = /^[^ ]+@[^ ]+\.[a-z]{2,3}$/;

    if (pattern.test(email)) {
        document.getElementById("emailResult").innerText = "Valid Email ✅";
    } else {
        document.getElementById("emailResult").innerText = "Invalid Email ❌";
    }
}

// Password strength
function checkPassword() {
    let pass = document.getElementById("password").value;
    let strength = 0;

    if (pass.length > 6) strength++;
    if (/[A-Z]/.test(pass)) strength++;
    if (/[0-9]/.test(pass)) strength++;
    if (/[@#$%^&*]/.test(pass)) strength++;

    let result = "";

    if (strength <= 1) result = "Weak ❌";
    else if (strength == 2) result = "Medium ⚠️";
    else result = "Strong ✅";

    document.getElementById("passResult").innerText = result;
}

// URL phishing detection
function checkURL() {
    let url = document.getElementById("url").value;
    let score = 0;

    if (url.length > 75) score++;
    if (url.includes("@")) score++;
    if (/(\d{1,3}\.){3}\d{1,3}/.test(url)) score++;
    if (!url.startsWith("https")) score++;

    let result = score >= 2 ? "Phishing ⚠️" : "Safe ✅";

    document.getElementById("urlResult").innerText = result;
}
