// Scene
const scene = new THREE.Scene();

// Camera
const camera = new THREE.PerspectiveCamera(
    75,
    window.innerWidth / window.innerHeight,
    0.1,
    1000
);
camera.position.z = 5;

// Renderer
const renderer = new THREE.WebGLRenderer({
    canvas: document.querySelector('#bg')
});
renderer.setSize(window.innerWidth, window.innerHeight);

// Light
const light = new THREE.PointLight(0xffffff, 1);
light.position.set(5, 5, 5);
scene.add(light);

// Bat Body
const bodyGeometry = new THREE.BoxGeometry(0.3, 0.6, 0.2);
const material = new THREE.MeshStandardMaterial({ color: 0x5500aa });
const body = new THREE.Mesh(bodyGeometry, material);
scene.add(body);

// Wings
const wingGeometry = new THREE.BoxGeometry(1, 0.1, 0.3);

const leftWing = new THREE.Mesh(wingGeometry, material);
leftWing.position.set(-0.8, 0, 0);
scene.add(leftWing);

const rightWing = new THREE.Mesh(wingGeometry, material);
rightWing.position.set(0.8, 0, 0);
scene.add(rightWing);

// Animation variables
let angle = 0;

// Animation loop
function animate() {
    requestAnimationFrame(animate);

    angle += 0.1;

    // Wing flapping
    leftWing.rotation.z = Math.sin(angle) * 0.5;
    rightWing.rotation.z = -Math.sin(angle) * 0.5;

    // Flying motion
    body.position.x += 0.02;
    leftWing.position.x = body.position.x - 0.8;
    rightWing.position.x = body.position.x + 0.8;

    // Up-down motion
    let y = Math.sin(body.position.x * 2) * 0.5;
    body.position.y = y;
    leftWing.position.y = y;
    rightWing.position.y = y;

    // Reset position
    if (body.position.x > 5) {
        body.position.x = -5;
    }

    renderer.render(scene, camera);
}

animate();
